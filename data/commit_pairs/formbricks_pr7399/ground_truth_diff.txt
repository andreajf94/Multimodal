+  const router = useRouter();
+
+  const { isOwner, isManager } = getAccessFlags(membershipRole);

+
+  const handleToggle = async (
+    field: "isAISmartToolsEnabled" | "isAIDataAnalysisEnabled",
+    checked: boolean

+    setIsLoading(true);

+      if (response?.data) {

+      }

+      setIsLoading(false);
+    }
+  };
